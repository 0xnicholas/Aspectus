Create `crates/aspectus-server/src/db/oauth_store.rs`:

```rust
use async_trait::async_trait;
use chrono::{DateTime, Utc};
use sqlx::PgPool;

use aspectus_core::error::CoreError;
use aspectus_core::store::{AuthorizationCodeStore, RefreshTokenStore, OAuth2ClientStore};

pub struct PgAuthorizationCodeStore { pool: PgPool }
pub struct PgRefreshTokenStore { pool: PgPool }
pub struct PgOAuth2ClientStore { pool: PgPool }

impl PgAuthorizationCodeStore { pub fn new(pool: PgPool) -> Self { Self { pool } } }
impl PgRefreshTokenStore { pub fn new(pool: PgPool) -> Self { Self { pool } } }
impl PgOAuth2ClientStore { pub fn new(pool: PgPool) -> Self { Self { pool } } }

#[async_trait]
impl AuthorizationCodeStore for PgAuthorizationCodeStore {
    async fn create_code(&self, code: &str, user_id: &str, client_id: &str,
        redirect_uri: &str, expires_at: DateTime<Utc>,
    ) -> Result<(), CoreError> {
        sqlx::query("INSERT INTO authorization_codes (code, user_id, client_id, redirect_uri, expires_at) VALUES ($1,$2,$3,$4,$5)")
            .bind(code).bind(user_id).bind(client_id).bind(redirect_uri).bind(expires_at)
            .execute(&self.pool).await.map_err(|e| CoreError::Internal(e.to_string()))?;
        Ok(())
    }

    async fn exchange_code(&self, code: &str) -> Result<Option<(String, String, String)>, CoreError> {
        sqlx::query_as::<_, (String, String, String)>(
            "UPDATE authorization_codes SET used = true WHERE code = $1 AND used = false AND expires_at > now() RETURNING user_id, client_id, redirect_uri"
        ).bind(code).fetch_optional(&self.pool).await
            .map_err(|e| CoreError::Internal(e.to_string()))
    }
}

#[async_trait]
impl RefreshTokenStore for PgRefreshTokenStore {
    async fn create(&self, token_hash: &str, user_id: &str, client_id: &str,
        expires_at: DateTime<Utc>) -> Result<(), CoreError> {
        sqlx::query("INSERT INTO refresh_tokens (token_hash, user_id, client_id, expires_at) VALUES ($1,$2,$3,$4)")
            .bind(token_hash).bind(user_id).bind(client_id).bind(expires_at)
            .execute(&self.pool).await.map_err(|e| CoreError::Internal(e.to_string()))?;
        Ok(())
    }

    async fn rotate(&self, token_hash: &str) -> Result<Option<(String, String, String)>, CoreError> {
        sqlx::query_as::<_, (String, String, String)>(
            "UPDATE refresh_tokens SET revoked_at = now() WHERE token_hash = $1 AND revoked_at IS NULL AND expires_at > now() RETURNING user_id, client_id, token_hash"
        ).bind(token_hash).fetch_optional(&self.pool).await
            .map_err(|e| CoreError::Internal(e.to_string()))
    }

    async fn revoke_all_for_user(&self, user_id: &str) -> Result<u64, CoreError> {
        let result = sqlx::query("UPDATE refresh_tokens SET revoked_at = now() WHERE user_id = $1 AND revoked_at IS NULL")
            .bind(user_id).execute(&self.pool).await
            .map_err(|e| CoreError::Internal(e.to_string()))?;
        Ok(result.rows_affected())
    }
}

#[async_trait]
impl OAuth2ClientStore for PgOAuth2ClientStore {
    async fn create(&self, client_id: &str, name: &str, redirect_uris: &[String]) -> Result<(), CoreError> {
        sqlx::query("INSERT INTO oauth2_clients (client_id, name, redirect_uris) VALUES ($1,$2,$3)")
            .bind(client_id).bind(name).bind(redirect_uris)
            .execute(&self.pool).await.map_err(|e| CoreError::Internal(e.to_string()))?;
        Ok(())
    }

    async fn list(&self) -> Result<Vec<(String, String, Vec<String>)>, CoreError> {
        sqlx::query_as("SELECT client_id, name, redirect_uris FROM oauth2_clients ORDER BY created_at DESC")
            .fetch_all(&self.pool).await
            .map_err(|e| CoreError::Internal(e.to_string()))
    }

    async fn validate_redirect_uri(&self, client_id: &str, redirect_uri: &str) -> Result<bool, CoreError> {
        sqlx::query_scalar::<_, bool>("SELECT EXISTS(SELECT 1 FROM oauth2_clients WHERE client_id = $1 AND $2 = ANY(redirect_uris))")
            .bind(client_id).bind(redirect_uri)
            .fetch_one(&self.pool).await
            .map_err(|e| CoreError::Internal(e.to_string()))
    }
}
```

Create `crates/aspectus-server/src/middleware/rate_limit.rs`:

```rust
use std::collections::HashMap;
use std::sync::Arc;
use std::time::{Duration, Instant};
use axum::{
    extract::Request, http::StatusCode, middleware::Next, response::{IntoResponse, Response},
};
use tokio::sync::Mutex;

/// Simple in-memory sliding-window rate limiter.
/// For production with multiple replicas, swap to Redis-backed.
#[derive(Clone)]
pub struct RateLimiter {
    inner: Arc<Mutex<HashMap<String, Vec<Instant>>>>,
    max_requests: usize,
    window: Duration,
}

impl RateLimiter {
    pub fn new(max_requests: usize, window_secs: u64) -> Self {
        Self {
            inner: Arc::new(Mutex::new(HashMap::new())),
            max_requests,
            window: Duration::from_secs(window_secs),
        }
    }

    async fn check(&self, key: &str) -> bool {
        let mut map = self.inner.lock().await;
        let now = Instant::now();
        let window_start = now - self.window;
        let timestamps = map.entry(key.to_string()).or_default();
        timestamps.retain(|t| *t > window_start);
        if timestamps.len() >= self.max_requests {
            return false;
        }
        timestamps.push(now);
        true
    }
}

pub async fn rate_limit_layer(
    limiter: RateLimiter,
    key_extractor: fn(&Request) -> String,
    mut req: Request,
    next: Next,
) -> Response {
    let key = key_extractor(&req);
    if !limiter.check(&key).await {
        let mut resp = axum::Json(serde_json::json!({
            "type": "https://aspectus.dev/errors/rate-limited",
            "title": "Too Many Requests",
            "status": 429,
            "detail": "Rate limit exceeded. Please retry after the window expires."
        })).into_response();
        *resp.status_mut() = StatusCode::TOO_MANY_REQUESTS;
        resp.headers_mut().insert("Retry-After", "60".parse().unwrap());
        return resp;
    }
    next.run(req).await
}

/// Extract client IP from request for rate limiting.
pub fn ip_key(req: &Request) -> String {
    req.headers().get("x-forwarded-for")
        .and_then(|v| v.to_str().ok())
        .and_then(|v| v.split(',').next())
        .map(|s| s.trim().to_string())
        .unwrap_or_else(|| "unknown".to_string())
}

/// Extract service token hash for management API rate limiting.
pub fn service_token_key(req: &Request) -> String {
    req.headers().get("authorization")
        .and_then(|v| v.to_str().ok())
        .and_then(|v| v.strip_prefix("Bearer "))
        .map(|s| s.to_string())
        .unwrap_or_else(|| "unknown".to_string())
}
```

- [ ] **Step 3: Refresh token reuse detection**

In `/token` refresh_token handler: after rotating the token, check if the old token was already revoked (i.e., `rotate()` returned `None`). If it returned `None` but the token was found in the DB with `revoked_at IS NOT NULL`, it means this is a **replay attack** — revoke ALL refresh tokens for that user:

```rust
"refresh_token" => {
    let hash = hex::encode(Sha256::digest(token.as_bytes()));
    match refresh_token_store.rotate(&hash).await {
        Ok(Some((user_id, client_id, _))) => {
            // Normal rotation — issue new tokens
            issue_tokens(&state, &user_id, &tenant, &client_id).await
        }
        Ok(None) => {
            // Token not found or already revoked
            // Check if it was explicitly revoked (replay attack)
            // If so, revoke all tokens for that user
            let count = refresh_token_store.revoke_all_for_user(&user_id).await?;
            if count > 0 {
                tracing::warn!(user_id = %user_id, revoked = count, "Refresh token replay detected — revoked all tokens");
            }
            ProblemDetails::unauthorized("Invalid or expired refresh_token", "/token")
        }
        Err(e) => ProblemDetails::from(e),
    }
}
```

use std::sync::Arc;
use once_cell::sync::Lazy;
use testcontainers_modules::postgres::Postgres;
use testcontainers_modules::redis::Redis;
use testcontainers::runners::SyncRunner;
use axum::{Router, body::Body};
use tower::ServiceExt;
use http_body_util::BodyExt;

/// Spin up PG + Redis via testcontainers, run migrations, seed service token, return test app.
pub struct TestHarness {
    pub pg_container: Postgres,
    pub redis_container: Redis,
    pub app: Router,
    pub db_url: String,
    pub redis_url: String,
    pub service_token: String,
    /// API Key created during setup for introspect tests.
    pub test_api_key: Option<super::introspect_test::TestApiKey>,
    pub test_user: Option<super::oauth_test::TestUser>,
}

impl TestHarness {
    pub async fn new() -> Self {
        // Start containers
        let pg = Postgres::default().start().unwrap();
        let redis = Redis::default().start().unwrap();

        let db_url = format!(
            "postgresql://postgres:postgres@{}:{}/postgres",
            pg.get_host().unwrap(), pg.get_host_port_ipv4().unwrap()
        );
        let redis_url = format!("redis://{}:{}", redis.get_host().unwrap(), redis.get_host_port_ipv4().unwrap());

        // Run migrations
        let pool = sqlx::PgPool::connect(&db_url).await.unwrap();
        sqlx::migrate!("../../migrations").run(&pool).await.unwrap();

        // Seed service token
        let svc_token = "test-service-token";
        let token_hash = hex::encode(sha2::Sha256::digest(svc_token.as_bytes()));
        sqlx::query("INSERT INTO service_tokens (project, token_hash) VALUES ($1, $2) ON CONFLICT DO NOTHING")
            .bind(aspectus_core::project::Project::Pandaria).bind(&token_hash)
            .execute(&pool).await.unwrap();

        // Create app (refactored from main.rs for test use)
        let app = build_test_app(pool.clone(), &redis_url).await;

        TestHarness {
            pg_container: pg, redis_container: redis, app, db_url, redis_url,
            service_token: svc_token.to_string(),
            test_api_key: None, test_user: None,
        }
    }
}
